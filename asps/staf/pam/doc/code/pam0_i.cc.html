{
PAM0_FT pam0_call = pam0_;
  printf("pam0_load_ami: Starting ####");
  printf("####################\n");
  STRING_SEQ_T specs;
  specs._length = specs._maximum = PAM0_RANK;
  specs._buffer = new char*[PAM0_RANK];

  specs._buffer[0] = new char[strlen(TB1_SPEC)+1];
  strcpy(specs._buffer[0],TB1_SPEC);
  specs._buffer[1] = new char[strlen(TB2_SPEC)+1];
  strcpy(specs._buffer[1],TB2_SPEC);

  broker->newInvoker("pam0",PAM0_RANK
               ,(FNC_PTR_T)pam0_call ,specs);
  for( int i=0;i<specs._maximum;i++ ){
     delete specs._buffer[i];
  }
  delete[] specs._buffer;
  printf("pam0_load_ami: Normal sucessful conclusion ######");
  printf("\n");
  return TRUE;
}
