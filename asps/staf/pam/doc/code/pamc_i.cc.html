STAFCV_T pamc_load_ami(amiBroker *broker)
{
PAMC_FT pamc_call = pamc_;
  printf("pamc_load_ami: Starting ####");
  printf("####################\n");
  STRING_SEQ_T specs;
  specs._length = specs._maximum = PAMC_RANK;
  specs._buffer = new char*[PAMC_RANK];

  specs._buffer[0] = new char[strlen(SCALARS_SPEC)+1];
  strcpy(specs._buffer[0],SCALARS_SPEC);
  specs._buffer[1] = new char[strlen(VECTORS_SPEC)+1];
  strcpy(specs._buffer[1],VECTORS_SPEC);

  broker->newInvoker("pamc",PAMC_RANK
               ,(FNC_PTR_T)pamc_call ,specs);
  for( int i=0;i<specs._maximum;i++ ){
     delete specs._buffer[i];
  }
  delete[] specs._buffer;
  printf("pamc_load_ami: Normal sucessful conclusion ######");
  printf("\n");
  return TRUE;
}
