#ifdef __cplusplus
#ifndef ASUHASHTABLE_H
#define ASUHASHTABLE_H

struct hashEntry {
	unsigned long key;	/* hash key */
	void	*value;
	hashEntry *next, *prev;
	hashEntry(unsigned long newkey)
		{key=newkey; next=0; prev=0;}
	~hashEntry()
	{
		if(next) next->prev = prev;
		if(prev) prev->next = next;
	}
};

typedef hashEntry* pHashEntry;

const NBUCKETS=4096;

class hashTable {
public:
	hashTable();
	~hashTable();
	void add(unsigned long key, void *value);
	void *lookup(unsigned long key);
	void remove(unsigned long key);
	void mapcar(void (*f) (pHashEntry));
private:
	pHashEntry bucket[NBUCKETS];
	
	pHashEntry addElement(unsigned long key);
	pHashEntry searchBucketFor(pHashEntry p, unsigned long key);
	
	hash(unsigned long);
};

#endif /*ASUHASHTABLE_H*/
#endif /*__cplusplus*/
