//:DESCRIPTION:	Initialize ASU
//:ARGUMENTS:	-- NONE --
//:RETURN VALUE:-- NONE --
//:<--------------------------------------------------------------------
int asu_init()
{
   EML_MESSAGE("Initializing ASU.");

/*- Setup basic KUIP commands. -*/
   asu_kuip_init_();

/*- Define the ASU KUIP commands. -*/
   asu_def_();

   return TRUE;        //OVERLOAD CALL: TRUE: stafGeneric.h(?), sutLib.cc(?)
}
