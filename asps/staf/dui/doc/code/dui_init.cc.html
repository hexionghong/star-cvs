   dui_def_();

   return TRUE;
}

//:>--------------------------------------------------------------------
//:ROUTINE:	int dui_start()
//:DESCRIPTION:	Start DUI
//:ARGUMENTS:	-- NONE --
//:RETURN VALUE:-- NONE --
//:<--------------------------------------------------------------------
int dui_start()
{
   EML_MESSAGE("Starting DUI.");

/*- Create the DUI Dispatcher. -*/
   dui = new duiDispatcher("dui");        //OVERLOAD CALL: duiDispatcher: duiClasses.cc(?), duiClasses.cc(?)

/*- Superceed the TDM Factory. -*/
   if( tdm != NULL ){
      EML_MESSAGE("DUI superceeds TDM.");
      tdm->lock(FALSE);
      soc->deleteObject("tdm","tdmFactory");
      tdm = dui;
   }

   return TRUE;
}
